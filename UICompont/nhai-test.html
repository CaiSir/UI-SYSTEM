🧪 测试代码

// 测试代码 console.log('开始测试 NHAIObjectFactory...'); // 检查 NHAIObjectFactory 是否可用 if (typeof NHAIObjectFactory !== 'undefined') { console.log('✅ NHAIObjectFactory 已定义'); // 创建容器 const container = NHAIObjectFactory.createContainer(); console.log('✅ 容器创建成功:', container); // 创建按钮 const button = NHAIObjectFactory.createButton('测试按钮'); button.setVariant('primary'); button.setOnClick(() => { console.log('✅ 按钮点击事件触发'); alert('测试成功！NHAIObjectFactory 工作正常'); }); // 添加到容器 container.addChild(button); console.log('✅ 按钮添加到容器'); // 渲染 const element = container.render(); console.log('✅ 渲染成功:', element); // 添加到预览区域 const previewArea = document.querySelector('.preview-area'); if (previewArea) { previewArea.appendChild(element); console.log('✅ 组件已添加到预览区域'); } console.log('🎉 所有测试通过！'); } else { console.error('❌ NHAIObjectFactory 未定义'); }
